cmake_minimum_required(VERSION 3.0 )

# project name
project( nccl_example LANGUAGES CXX CUDA )
message(STATUS "CMAKE_CUDA_COMPILER ${CMAKE_CUDA_COMPILER}")

# find nccl
list(APPEND CMAKE_MODULE_PATH ${PROJECT_SOURCE_DIR}/cmake/)

# nccl
find_package(nccl REQUIRED)
if ( NCCL_FOUND )
    message(STATUS "!!!!NCCL Found ${NCCL_INCLUDE_DIRS} ${NCCL_LIBRARIES}!!!")
else()
    message(STATUS "!!!!NCCL Not Found!!")
endif()

# header
include_directories( ${NCCL_INCLUDE_DIRS} )

# exec
add_executable( demo_group0123 demo_group0123.cu  )

# I'm using C++11 in the test
target_compile_features( demo_group0123 PRIVATE cxx_std_11)

# link nccl
target_link_libraries( demo_group0123 PRIVATE ${NCCL_LIBRARIES} )
